2 — Energy Model Eq.7–8 · หน้า 621
eorl.pyEq.7–8
# Eq.7: Etx(bl, d) — พลังงานส่ง ขึ้นกับระยะทาง d
def etx(d):
    if d <= d0:  # free space: ใช้ d²
        return bl * (Efs * d**2 + Eelec)
    else:       # multipath: ใช้ d⁴ (กินมากกว่า)
        return bl * (Emp * d**4 + Eelec)

# Eq.8: Erx = bl × Eelec (ไม่ขึ้นกับระยะทาง)
def erx():
    return bl * Eelec
ทุก edge ใน network ระยะ < 277 ม. → ใช้ free space ทั้งหมด
etx(75m) = 800 × (100×10⁻¹² × 5625 + 50×10⁻⁹) = 0.000490 J  |  erx() = 800 × 50×10⁻⁹ = 0.000040 J
3 — Generate Spanning Trees Section III-C · หน้า 620
eorl.pyAction Space
# paper: "action space = all possible routes via spanning tree algorithm"
# AS = {r1, r2, ..., rm}  →  m = 4,469 trees
def generate_spanning_trees(graph):
    nodes = list(graph.nodes)    # 9 nodes
    edges = list(graph.edges)    # 17 edges
    trees = []

    # C(17, 8) = 24,310 subsets → ตรวจ is_tree → ได้ 4,469
    for subset in combinations(edges, len(nodes)-1):
        g2 = nx.Graph()
        g2.add_nodes_from(nodes)
        g2.add_edges_from(subset)
        if nx.is_tree(g2):     # connected + ไม่มีวงวน
            trees.append(g2)
    return trees
4 — State Function st(min) Eq.2 · หน้า 620
eorl.pyEq.2
# Eq.2: st(min) = min_{i=1}^{n} [ERi(st-1, at-1)]
# แปลงเป็น index 0–500 สำหรับ row ของ Q-table
def get_state(energy, active):
    if not active: return 0
    v = max(0.0, min(energy[n] for n in active))
    return min(int(v / STATE_STEP), N_STATES-1)
ตัวอย่าง: energy = [-, 3.50, 2.36, 4.18, ...] → min = 2.36 J → state = 236
5 — Reward Function Rmin Eq.6 · หน้า 621
eorl.pyEq.6
# Eq.6: Rmin(at) = min_i( ERi(st,at) / Einit )
# reward = normalized energy ของ node ที่แย่สุด (0–1)
def reward_min(energy, active):
    if not active: return 0.0
    return min(max(0.0, energy[n]) / Einit
               for n in active)
ตัวอย่าง: min energy = 2.36 J → Rmin = 2.36/5.0 = 0.472 Rmin สูง = node แย่สุดยังมีพลังงานดี → tree นี้ balance → RL เลือกบ่อยขึ้น

for rnd in range(1, MAX_ROUNDS+1):

    # หยุดเมื่อไม่มี bottleneck node เหลือเลย
    if not active or not (active & BOTTLENECK): break

    # ── Step 1: State ปัจจุบัน (Eq.2) ──────────────────
    s = get_state(energy, active)

    # ── Step 2: ε-greedy เลือก spanning tree ───────────
    if random.random() < epsilon:   # 10%: explore
        action = random.randint(0, N_TREES-1)
        else:                          # 90%: exploit
        action = int(np.argmax(Q[s]))

    # ── Step 3: Route + simulate (Eq.7–8) ──────────────
    routes = tree_to_routes(trees[action], active)
    if not routes: routes = sdwsn_routing(g, active)

    # ── Step 4: Reward Rmin (Eq.6) ─────────────────────
    r = reward_min(energy, active)

    # ── Step 5: อัปเดต active set ──────────────────────
    dead = {n for n in active if energy[n] <= 0}
    active -= dead

    Q[s, action] = (1-alpha)*Q[s,action] + alpha*(r + gamma*np.max(Q[s2]))

7 — simulate_round Eq.7–8 applied · หน้า 621

def simulate_round(routes, energy, graph):
    for node, path in routes.items():
        for i in range(len(path)-1):
            sender, receiver = path[i], path[i+1]
            d = graph[sender][receiver].get('distance', 50)
            energy[sender]   -= etx(d)  # Eq.7
            energy[receiver] -= erx()  # Eq.8
